class MyStack:

    def __init__(self):
        self.arr=[]

    def push(self, x: int) -> None:
        self.arr.append(x)

    def pop(self) -> int:
        return self.arr.pop()

    def top(self) -> int:
        lenOfarr=len(self.arr)
        return self.arr[lenOfarr-1]

    def empty(self) -> bool:
        if len(self.arr)==0:
              return True
        else:
              return False
    def size(self) -> int:
        return len(self.arr)
